[System]
StreamMemoryMB   = 2048      	  # Менеджер стрим-памяти. Значение - объем ОЗУ (в Мб), отводимое для игры. Максимальное значение - 2048 Мб. Стандартное значение: 50 Мб. По сути, эта функция - тот же самый ASI-плагин MemStreamFix от Alexander Blade / CLEO-версия Stream Memory Fix от fastman92 с исправлениями от LINK/2012. Функция доступна также в Open Limit Adjuster от LINK/2012 & 13AG или в Fastman Limit Adjuster от fastman92.
FPSLimit         = AUTO       # Ограничитель количества кадров в секунду. Стандартное значение - 30. Увеличение вызовет баги, типа невозможности целиться из снайперки/дигла, к срыву концовок кат-сцен, невозможности нырять и т.п. Значение "AUTO" отдаст определение лимита на откуп самой игре (при этом в ситуации, когда лимит может быть безболезненно повышен до 60, он будет повышен. Сам). Сия функция работает, если ограничитель кадров включен в меню игры! Для фикса некоторых багов, связанных с увеличенным значением FPS, используйте мод http://bit.do/fix-aim-rifle-walk  (Автор: Junior_Djjr)
FrameDelay       = 0          # Задержка между выводом очередного кадра (FPS delay). При установке значения "0" количество кадров в секунду возрастет примерно на 20%. Стандартное значение: 14. По сути, функция была применена еще в Stream INI Extender от Deji. Доступна также в SilentPatch.
ProcessPriority  = 4          # Приоритет процесса gta_sa.exe (от 1 до 6). Стандартное значение - 3. Чем выше значение, тем выше будет приоритет процесса для ЦП вашего ПК. Не используйте "6" если ЦП вашего ПК имеет одно ядро. Для изменения процесса прямо в игре, воспользуйтесь комбинацией клавиш "INSERT" + "1...6" (в зависимости от желаемого приоритета). (Автор: Fabio)
AntiAliasing     = -1         # Настройка используемого уровня сглаживания. Значение может принимать значения от "0" (выкл.) до максимального, поддерживаемого вашей видеокартой. При назначении слишком большого значения, оно будет сброшено до максимального, которое поддерживает ваш графический адаптер. Во время игры введите код "CA#A", где # - цифра, соответствующая желаемому уровню сглаживания. Кодом "AAI" можно проверить максимальный уровень, поддерживаемый вашей видеокартой. Важно: если вы используете модификации Shell или Colormod от Ryosuke, то функция смены уровня сглаживания в игре у вас работать не будет. Оригинальный мод - Multi-Sampling Fix от DK22Pac (http://gtaforums.com/topic/556386-relsa-multi-sampling-fix/)
